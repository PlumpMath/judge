enum resultValues {
    QUE = 0,  // Queued
    CMP = 1,  // Compiling
    JUD = 2,  // Judging
    CE = 3,   // Compile Error
    RE = 4,   // Runtime Error
    ME = 5,   // Memory Exceeded
    WA = 6,   // Wrong Answer
    TLE = 7,  // Time Limit Exceeded
    OLE = 8,  // Output Limit Exceeded
    SE = 9,   // System Error (IO, ..)
    AC = 10   // Accepted
};

CREATE DATABASE `judge`;

CREATE TABLE `judge`.`problems`(
  `id` INT NOT NULL AUTO_INCREMENT,
  `title` VARCHAR(200) NOT NULL,
  `submit_count` INT NOT NULL DEFAULT '0',
  `accept_count` INT NOT NULL DEFAULT '0',
  `accept_users` INT NOT NULL DEFAULT '0',
  `description` TEXT NULL DEFAULT NULL,
  `hint` TEXT NULL DEFAULT NULL,
  `case_count` INT NOT NULL DEFAULT '3',
  `added_date` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `source` VARCHAR(500) DEFAULT 'None',
  `time_limit` INT NOT NULL DEFAULT '1',
  `memory_limit` INT NULL DEFAULT '256',
  `sample_input` TEXT NULL DEFAULT NULL,
  `sample_output` TEXT NULL DEFAULT NULL,
  PRIMARY KEY(`id`)
) ENGINE = InnoDB DEFAULT CHARSET=utf8;
